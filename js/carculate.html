function count(){ var tx1=document.getElementById("txt1").value; var chart=document.getElementById("select").value; var tx2=document.getElementById("txt2").value; var result=""; switch(chart) {case "+": result=parseFloat(tx1)+parseFloat(tx2); break; case "-": result=parseFloat(tx1)-parseFloat(tx2); break; case "*": result=parseFloat(tx1)*parseFloat(tx2); break; default: result=parseFloat(tx1)/parseFloat(tx2); break; } document.getElementById("fruit").value=result; }
